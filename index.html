            { id: 2, name: "Quality Hand Grinder", price: 150.00, img: "img/grinder.jpg" },

            { id: 7, name: "Precision Coffee Scale", price: 280.00, img: "img/premium_scale.jpg" },

            { id: 6, name: "Entry Espresso Machine", price: 350.00, img: "img/espresso.jpg" },

        ].sort((a, b) => a.price - b.price); // Sort by price ascending


        // 2. DOM Elements & Chart Setup

        const dailySpendInput = document.getElementById('daily-spend');

        const numberOfDaysInput = document.getElementById('number-of-days');

        const noProductsMessage = document.getElementById('no-products');

        

        // Initialize Chart

        const ctx = document.getElementById('savings-chart').getContext('2d');

        const savingsChart = new Chart(ctx, {

            type: 'line', // Simple line chart for progress

            data: {

                labels: [], // Days will go here

                datasets: [{

                    fill: true,

                    pointBackgroundColor: 'rgb(255, 107, 246)' // Neon Magenta points

                }]

            },

            options: {

        // 3. Core Calculation & Graph Update Function

        function calculateSavingsAndDisplay() {

            const dailySpend = parseFloat(dailySpendInput.value) || 0;

            const numberOfDays = parseInt(numberOfDaysInput.value) || 0;


            const totalSaved = dailySpend * numberOfDays;

            

            // 3a. Update Total Savings Display

            const formattedTotal = new Intl.NumberFormat('en-US', {

                style: 'currency',

                currency: 'USD'

            }).format(totalSaved);

            totalSavedSpan.textContent = formattedTotal;

            

            // 3b. Update Chart Data

            updateGraph(dailySpend, numberOfDays);


            // 3c. Render Products

            renderAffordableProducts(totalSaved);

        }

        

        function updateGraph(dailySpend, days) {

            let labels = [];

            let data = [];

            

            // Only plot key points for a simple UX-focused graph

            const intervals = [1, 7, 14, 30, 60, 90, 180]; 

            

            for (const interval of intervals) {

                if (interval <= days) {

                    labels.push(`Day ${interval}`);

                    data.push(dailySpend * interval);

                }

            }

            

            // Ensure the final day is included if it's not a standard interval

            if (days > 0 && !intervals.includes(days)) {

                 labels.push(`Day ${days}`);

                 data.push(dailySpend * days);

            }


            savingsChart.data.labels = labels;

            savingsChart.data.datasets[0].data = data;

            savingsChart.update();

        }


        // 4. Product Rendering Function

        function renderAffordableProducts(maxPrice) {

            const affordableProducts = products.filter(product => product.price <= maxPrice);


            productCarousel.innerHTML = '';


            if (affordableProducts.length > 0) {

                noProductsMessage.style.display = 'none';

                

                affordableProducts.forEach(product => {

                    const card = document.createElement('div');

                    card.className = 'product-card';

                    card.innerHTML = `

                        <img src="${product.img}" alt="${product.name} (Image Placeholder)">

                        <h3>${product.name}</h3>

                        <p class="price">${new Intl.NumberFormat('en-US', { style: 'currency', currency: 'USD' }).format(product.price)}</p>

                        <a href="#" class="affiliate-link" target="_blank">View & Buy Now</a>

                    `;

                    productCarousel.appendChild(card);

                });

            } else {

                noProductsMessage.style.display = 'block';

            }

        }


        // 5. Attach Event Listeners

        dailySpendInput.addEventListener('input', calculateSavingsAndDisplay);

        numberOfDaysInput.addEventListener('input', calculateSavingsAndDisplay);

        calculateSavingsAndDisplay();
